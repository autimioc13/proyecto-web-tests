- SendGrid API (transactional emails)

- TypeScript with strict types

```typescript
import crypto from 'crypto';

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY;
const ENCRYPTION_IV = process.env.ENCRYPTION_IV;

if (!ENCRYPTION_KEY || !ENCRYPTION_IV) {
  console.warn('⚠️ ENCRYPTION_KEY or ENCRYPTION_IV not set. Encryption will fail.');
}

// Convert base64 strings back to buffers
function getKeyBuffer(): Buffer {
  if (!ENCRYPTION_KEY) throw new Error('ENCRYPTION_KEY not configured');
  return Buffer.from(ENCRYPTION_KEY, 'base64');
}

function getIvBuffer(): Buffer {
  if (!ENCRYPTION_IV) throw new Error('ENCRYPTION_IV not configured');
  return Buffer.from(ENCRYPTION_IV, 'base64');
}

/**
 * Encrypt sensitive data using AES-256-GCM
 * Returns: iv:authTag:encryptedData (all hex-encoded)
 */
export function encryptSensitive(plaintext: string): string {
  try {
    const key = getKeyBuffer();
    const iv = getIvBuffer();

    const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');

    const authTag = cipher.getAuthTag();

    // Format: iv:authTag:encrypted
    return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
  } catch (error) {
    console.error('[ENCRYPTION ERROR]', error);
    throw error;
  }
}

/**
 * Decrypt sensitive data
 * Input format: iv:authTag:encryptedData
 */
export function decryptSensitive(encrypted: string): string {
  try {
    const [ivHex, authTagHex, encryptedHex] = encrypted.split(':');

    if (!ivHex || !authTagHex || !encryptedHex) {
      throw new Error('Invalid encrypted data format');
    }

    const key = getKeyBuffer();
    const iv = Buffer.from(ivHex, 'hex');
    const authTag = Buffer.from(authTagHex, 'hex');

    const decipher = crypto.createDecipheriv('aes-256-gcm', key, iv);
    decipher.setAuthTag(authTag);

    let decrypted = decipher.update(encryptedHex, 'hex', 'utf8');
    decrypted += decipher.final('utf8');

    return decrypted;
  } catch (error) {
    console.error('[DECRYPTION ERROR]', error);
    throw error;
  }
}

/**
 * Hash a value (one-way, for pseudonymization)
 */
export function hashValue(value: string): string {
  return crypto.createHash('sha256').update(value).digest('hex');
}

/**
 * Generate a secure random token (for deletion confirmations, etc)
 */
export function generateSecureToken(length: number = 32): string {
  return crypto.randomBytes(length).toString('hex');
}

/**
 * Test encryption/decryption
 */
export function testEncryption(): boolean {
  try {
    const testData = 'test-sensitive-data-12345';
    const encrypted = encryptSensitive(testData);
    const decrypted = decryptSensitive(encrypted);
    const success = decrypted === testData;

    if (success) {
      console.log('✓ Encryption/decryption test passed');
    } else {
      console.error('✗ Decrypted data does not match original');
    }

    return success;
  } catch (error) {
    console.error('✗ Encryption test failed:', error);
    return false;
  }
}
```

    throw error;
  }

 * Get all activity logs for a user (for GDPR export)
 */
export async function getUserActivityLogs(userId: string) {
  return prisma.activityLog.findMany({
    where: { userId },
    orderBy: { createdAt: 'desc' },
  });

/**
 * Get activity logs for a specific resource
 */
export async function getActivityLogsByResource(resource: string, limit: number = 100) {
  return prisma.activityLog.findMany({
    where: { resource },
    orderBy: { createdAt: 'desc' },
    take: limit,
  });
}

/**
 * Get recent security events
 */
export async function getRecentSecurityEvents(hours: number = 24) {
  const since = new Date(Date.now() - hours * 60 * 60 * 1000);

  return prisma.activityLog.findMany({
    where: {
      activityType: 'security_event',
      createdAt: { gte: since },
    },
    orderBy: { createdAt: 'desc' },
  });
}

 * Clean up old activity logs (keep last 1 year by default)

  const cutoffDate = new Date(Date.now() - daysToKeep * 24 * 60 * 60 * 1000);

  const result = await prisma.activityLog.deleteMany({
    where: { createdAt: { lt: cutoffDate } },
  });

  console.log(`[CLEANUP] Deleted ${result.count} old activity logs`);
  return result;
}

/**
 * Confirm a deletion request (user clicked email link)
 */
export async function confirmDeletionRequest(token: string) {
  try {
    const request = await prisma.deletionRequest.findUnique({
      where: { confirmationToken: token },
    });

    if (!request) {
      throw new Error('Token not found or invalid');
    }

    if (new Date() > request.tokenExpiresAt) {
      throw new Error('Token has expired');
    }

    if (request.confirmed) {
      throw new Error('Deletion request already confirmed');
    }

    // Mark as confirmed
    const updated = await prisma.deletionRequest.update({
      where: { id: request.id },
      data: {
        confirmed: true,
        confirmedAt: new Date(),
        status: 'confirmed',
      },
    });

    console.log(`[DELETION CONFIRMED] ${request.email}`);
    return updated;
  } catch (error) {
    console.error('[CONFIRM DELETION ERROR]', error);
    throw error;
  }
}

export async function completeDeletion(email: string) {

      data: {

      },

 * Check if token is valid
 */
export async function isValidDeletionToken(token: string): Promise<boolean> {
  try {
    const request = await prisma.deletionRequest.findUnique({
      where: { confirmationToken: token },
    });

    if (!request) return false;
    if (request.confirmed) return false;
